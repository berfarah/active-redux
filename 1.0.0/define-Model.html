    Attributes of a model, for which methods will be defined
    Source:

    Example
    import Model, { Attr } from 'active-redux';

const Person = Model.define('people', class Person {
  static attributes = {
    name: Attr.string(),
  }
});

const joe = new Person({ attributes: { name: "Joe" } });
joe.name // => "Joe"
